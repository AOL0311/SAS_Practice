# PRactice

```sas
data df_iop;
	set df;

	if IOPL > 21 then DFIOPL = 1;
	else DFIOPL = 0;

	if IOPR > 21 then DFIOPR = 1;
	else DFIOPR = 0;

proc freq data = df_iop;
	tables DFIOPL DFIOPR;
run;

proc freq data = df_iop;
	tables sex * DFIOPL / Expected chisq;
proc freq data = df_iop;
	tables sex * DFIOPR / Expected chisq;
run;
```
